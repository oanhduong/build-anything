```python
"""Module 1: Excel ingestion and schema detection."""
from pathlib import Path
import pandas as pd


def ingest_excel(path: str | Path) -> tuple[dict, pd.DataFrame]:
    """Read Excel file and detect schema.

    Args:
        path: Path to .xlsx file

    Returns:
        Tuple of (schema_dict, dataframe)
        schema_dict keys: row_count (int), columns (dict of col -> {dtype, null_count, sample})
    """
    path = Path(path)
    # engine="openpyxl" is REQUIRED for .xlsx — xlrd dropped .xlsx support in pandas 1.2+
    df = pd.read_excel(path, engine="openpyxl")

    schema = {
        "row_count": len(df),
        "columns": {
            col: {
                "dtype": str(df[col].dtype),
                "null_count": int(df[col].isna().sum()),
                "sample": df[col].dropna().iloc[:3].tolist() if not df[col].dropna().empty else [],
            }
            for col in df.columns
        },
    }
    return schema, df
```

```python
"""Module 2: Schema normalization."""
import pandas as pd


def normalize(schema: dict, df: pd.DataFrame) -> dict:
    """Map detected schema into a normalized structure.

    Args:
        schema: Output dict from ingest_excel()
        df: DataFrame from ingest_excel()

    Returns:
        Normalized structure with clean column names, null percentages, sample rows
    """
    total_rows = schema["row_count"]

    columns = [
        {
            "name": col,
            "dtype": info["dtype"],
            "null_pct": round(info["null_count"] / total_rows * 100, 2) if total_rows > 0 else 0.0,
        }
        for col, info in schema["columns"].items()
    ]

    # Convert up to 3 rows to dicts; replace NaN with None for JSON safety
    sample_rows = df.head(3).where(df.head(3).notna(), other=None).to_dict(orient="records")

    return {
        "row_count": total_rows,
        "columns": columns,
        "sample_rows": sample_rows,
    }
```

```python
"""Module 3: AI analysis and recommendations via Anthropic Claude API."""
import anthropic
from dotenv import load_dotenv

# load_dotenv() must be called HERE (at module level), not only in main.py
# This ensures ANTHROPIC_API_KEY is available when pytest imports this module
load_dotenv()


def get_recommendations(normalized_data: dict) -> str:
    """Analyze normalized data and return AI recommendations.

    Args:
        normalized_data: Output dict from normalize()

    Returns:
        Non-empty string with 3 recommendations from Claude Haiku
    """
    client = anthropic.Anthropic()  # reads ANTHROPIC_API_KEY from env (loaded above)

    prompt = (
        f"Analyze this dataset schema and provide exactly 3 actionable recommendations "
        f"for data quality improvement or analysis:\n\n{normalized_data}"
    )

    message = client.messages.create(
        model="claude-haiku-4-5",   # Haiku: fast + cheap for PoC (Claude's Discretion)
        max_tokens=1024,
        messages=[{"role": "user", "content": prompt}],
    )
    return message.content[0].text
```
